```json
{
  "sources": [
    {
      "id": "source-id",
      "path": "/points.geojson",
      "type": "geojson"
    }
  ],
  "layers": [
    {
      "id": "layer-id",
      "title": "Коммерция",
      "visualizations": [
        {
          "id": "vis-id",
          "mapLayerProps": {
            "type": "circle",
            "paint": {
              "circle-color": "#93fa16",
              "circle-stroke-width": 1,
              "circle-radius": 5
            }
          }
        }
      ]
    }
  ]
}
```
